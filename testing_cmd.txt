# What else got injected? Compare process tree against home
ps -ef | awk '{print $8}' | sort -u > /tmp/work-procs.txt

# Open file handles on overlay paths from non-container processes
sudo lsof 2>/dev/null | grep -E 'overlay|svc_heimdall' | grep -vE 'podman|conmon|crun|nginx|postgres'

# inotify saturation (CB and similar agents eat watches aggressively)
sudo find /proc/*/fd -lname 'anon_inode:inotify' 2>/dev/null | \
    awk -F/ '{print $3}' | sort | uniq -c | sort -rn | head

cat /proc/sys/kernel/tainted

sudo dmesg -T | tail -50
