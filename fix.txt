# 1. Stop the workload containers cleanly (pod stays up)
sudo -i -u svc_heimdall env XDG_RUNTIME_DIR=/run/user/1001 \
  DBUS_SESSION_BUS_ADDRESS=unix:path=/run/user/1001/bus \
  systemctl --user stop heimdall-database heimdall-server heimdall-nginx heimdall-gateway

# 2. Reapply SELinux labels (idempotent if already correct)
sudo restorecon -RFv /home/svc_heimdall/.local/share

# 3. Re-own pgdata to in-container postgres UID 1001
sudo -i -u svc_heimdall env XDG_RUNTIME_DIR=/run/user/1001 \
  podman unshare chown -R 1001:1001 /home/svc_heimdall/.local/share/heimdall_pgdata

# 4. Re-own nginx_certs to in-container nginx UID 997, GID 995
sudo -i -u svc_heimdall env XDG_RUNTIME_DIR=/run/user/1001 \
  podman unshare chown -R 997:995 /home/svc_heimdall/.local/share/heimdall_nginx_certs

# 5. Verify
sudo ls -la /home/svc_heimdall/.local/share/ | grep heimdall_

# 6. Start database first and watch
sudo -i -u svc_heimdall env XDG_RUNTIME_DIR=/run/user/1001 \
  DBUS_SESSION_BUS_ADDRESS=unix:path=/run/user/1001/bus \
  systemctl --user start heimdall-database
sleep 10
sudo -i -u svc_heimdall env XDG_RUNTIME_DIR=/run/user/1001 podman ps -a

# 7. If database is healthy, bring up the rest
sudo -i -u svc_heimdall env XDG_RUNTIME_DIR=/run/user/1001 \
  DBUS_SESSION_BUS_ADDRESS=unix:path=/run/user/1001/bus \
  systemctl --user start heimdall-server heimdall-nginx heimdall-gateway
